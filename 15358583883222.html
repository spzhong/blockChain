要想了解比特币，首先要知道是如何交易的，以及底层的结构和数据脚本是如何实现的；
本文主要是对比特币交易系统的分析；

整个交易的过程中，其实最难理解是索引和输入输出，接下来我分别介绍一下

  • 输出：通常理解为锁定脚本，决定谁可以花费这币交易
  • 输入：通常理解为解锁脚本，证明我已经花费这币交易
  • UTXO：未花费交易，获取输出脚本为当前钱包地址的所有交易信息列表，然后加上索引，匹配所有交易的输入是否被引用，否的话称为“UTXO”
  • 索引：在一个交易中，可能会被分解为多个UTXO,就是通过索引来确定该交易属于那个钱包地址的UTXO

2.首先要知道钱包是如何创建的

私钥                  随机32位字符（256位随机数）
公钥                  椭圆曲线加密SECP2561算法
公钥哈希值           SHA256(公钥哈希)
公钥哈希值              Ripemd160(公钥哈希值)
校验码                SHA256(SHA256(1个字节版本号0x00+公钥哈希值))，取前4个字节
钱包地址              base58(1个字节版本号0x00+公钥哈希值+校验码)
例如:1LYBoAY8u4sCrNEpVoB3SxkoKDm8aAF5yz

3.了解比特币涉及到的块，交易的数据结构定义

#块的结构
type Block struct {
        hash               string       哈希
        confirmations      string       确认次数
        size               uint64       块的大小
        height             uint64       块的高度
        version            uint32       块的版本
        merkleroot         string       默克尔树（后期讲解）
        tx                 []           交易的详细信息集合
        time               uint32       块的时间
        nonce              uint32       随机数（最大2^32，后期讲解）
        bits               string       当前目标hash值
        difficulty         float64      难度系数（用于挖矿控制，后期讲解）
        previousblockhash  string       上一个区块的hash
}

#交易的结构
type tx struct {
        hex                 string              交易的十六进制
        txid                string              交易的id
        version             uint64              版本号
        locktime            uint64              锁定时间（后期在脚本中会讲解）
        vin                 []                  交易的输入
        vout                []                  交易的输出
        blockhash           string              所在的块的hash
        confirmations       uint32              交易确认的次数
        time                uint32              交易的时间
        blocktime           uint32              所在快的时间
}

#交易的输入（挖矿所得的币）结构
type vin_coinbase struct {
        coinbase            string              挖矿所得的备注说明
        sequence            uint32              序列
}

#交易的输入（交易所得的币）结构
type vin_tx struct {
        txid                string             交易的id
        vout                uint32             交易的索引（和上个交易输出的vout中n对应）
        sequence            uint32             输入序列（后期在脚本中会讲解）
        scriptSig           map                解锁脚本
        {   
            asm => "私钥签名",
            hex => "16进制"
        }
}

#交易的输出结构
type vout struct {
        value               uint32           交易金额
        n                   uint32           交易引用索引           
        scriptPubKey        map              锁定脚本 
        {
            asm => "脚本code",
            hex => "签名验证16进制",
            reqSigs => "签名个数，默认至少为1",
            type => "pubkey:公钥验证，pubkeyhash:私钥签名验证",
            addresses => ["钱包的地址"]
        }                 
}
